#include <stdio.h>

int main(){
    int n=9;
    int r=n-1;
    int x=3;
    for(int i=1;i<=n/2+1;i++){
        for(int j=1;j<=r;j++){
            printf(" ");
        }
        if(i==1){
            printf("*");
        }
        else{
            for(int j=1;j<=x;j++){
                if(j==1 || j==x){
                    printf("* ");
                }
                else{
                    printf("  ");
                }
            }
            x=x+2;
        }
        r=r-2;
        printf("\n");
    }
    
    
    for(int i=1;i<=n/2;i++){
        for(int j=1;j<=i*2;j++){
            printf(" ");
        }
        if(i==n/2){
            printf("*");
        }
        else{
            for(int j=x-4;j>=1;j--){
                if(j==x-4 || j==1){
                    printf("* ");
                }
                else{
                    printf("  ");
                }
            }
            x=x-2;
        }
        printf("\n");
    }
    return 0;
}
