#include <stdio.h>

int main(){
    
    int left = 11;
    int right = 23;
    
    for(int i = left; i<=right;i++){
        for(int j=2;j<i;j++){
            if(i%j==0){
                break;
            }
            if((i-1) == j){
                printf("%d ",i);
            }
        }
    }
    
    return 0;
}
